def chebyshev_eval(coeffs, t_normalized):
    """Evaluate Chebyshev series at t ‚àà [-1, 1]"""
    T0 = 1.0
    T1 = t_normalized
    result = coeffs[0] * T0 + coeffs[1] * T1

    for i in range(2, len(coeffs)):
        T2 = 2 * t_normalized * T1 - T0
        result += coeffs[i] * T2

            vel1 * h * (h2 - h))
